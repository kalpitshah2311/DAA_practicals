#include <iostream>

using namespace std;
int cnt_recursion = 0;
int fib_using_recursion(int n)
{
    if(n==0)
    {
        cnt_recursion++;
    return 0;
    }
    else if(n==1)
    {
      cnt_recursion++;  
    return 1;
    }
    else
    {
       cnt_recursion++; 
    return fib_using_recursion(n-1)+fib_using_recursion(n-2);
    }
}

int fib_using_iteration(int n)
{ int count=0;
    if(n==0)
    return 0;
    else if(n==1)
    return 1;
    else
    {
        int first =0;
    int second =1;
    int ans=0;
    for(int i=2;i<=n;i++)
    {
        ans = first + second;
        count++;
        first = second;
        count++;
        second = ans;
        count++;
    }
    return count;
    }
    
}


int main()
{
    cout<<"cnt for recursion"<<endl;
    for(int i=3;i<=9;i++)
    {
        
  fib_using_recursion(i);
     cout<<"n = "<<i<<" cnt = "<<cnt_recursion<<endl;
    cnt_recursion = 0; // because cnt_recursion is global 
    }
    
    cout<<endl;
    cout<<"cnt for iterative"<<endl;
    for(int i=3;i<=9;i++)
    {
        
    
     cout<<"n = "<<i<<" cnt = "<<fib_using_iteration(i)<<endl;
   
    }
   
   return 0;
}
